package int101week03;

public class Circle {
    double redius, area, perimeter;

    public Circle(double redius) {
        this.area = 3.14 * Math.pow(redius, 2);
        this.perimeter = 2 * 3.14 * redius;
    }

    public double getArea() {
        return this.area;
    }

    public double getPerimeter() {
        return this.perimeter;
    }
}
